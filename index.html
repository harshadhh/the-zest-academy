} catch(err) { return out({ ok: false, error: 'Server error: ' + err.message }); } } function doGet(e) { return out({ ok: true, service: 'Zest Admin API', version: '2.0' }); } // ═══════════════════════════════════════════════════════════════════════ // AUTH — secure token-based sessions // ═══════════════════════════════════════════════════════════════════════ function doLogin({ hash }) { const stored = PropertiesService.getScriptProperties().getProperty('pw_hash') || DEFAULT_HASH; } // ═══════════════════════════════════════════════════════════════════════ // GET — read all rows of a collection as JSON array // ═══════════════════════════════════════════════════════════════════════ function doGet({ collection }) { if (!SHEET[collection]) return out({ ok: false, error: 'Unknown: ' + collection }); hdrs.forEach((h, i) => { o[h] = row[i]; }); return o; }); return out({ ok: true, rows }); } // ═══════════════════════════════════════════════════════════════════════ // SET — replace entire collection (full sync from localStorage) // ═══════════════════════════════════════════════════════════════════════ function doSet({ collection, rows, headers }) { if (!SHEET[collection]) return out({ ok: false, error: 'Unknown: ' + collection }); i++) tab.autoResizeColumn(i); } return out({ ok: true, written: rows ? rows.length : 0 }); } // ═══════════════════════════════════════════════════════════════════════ // APPEND — add single row (faster than full SET for one-off adds) // ═══════════════════════════════════════════════════════════════════════ function doAppend({ collection, row, headers }) { if (!SHEET[collection]) return out({ ok: false, error: 'Unknown: ' + collection });
